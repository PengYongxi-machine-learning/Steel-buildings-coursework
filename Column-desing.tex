  \item[]Class 1 limit for web, 72 $\epsilon $ = 22.78
  \item[]$c_{f}$ = $h_{w}$ = 87.70

\section{Buckling resistance about y-y
axis}\label{buckling-resistance-about-y-y-axis}

\begin{description}
  \item[]$L_{cr,y}$ = 6000 mm
  \item[]$N_{cr,y}$ = $\frac{\pi^{2}\cdot E\cdot I_{y}}{{L_{cr,y}}^{2}}$ = 8866.19 kN
  \item[]$\lambda_{y}$ =  $\sqrt{\frac{N_{Rk}}{N_{cr,y}}}$ = $\sqrt{\frac{A\cdot f_{y}}{N_{cr,y}}}$ = 0.81 --- (EN 1993-1-1 :2022, Eq 8.69)
  \item[]$\frac{h}{b}$ = 1.13 < 1.2
  \item[]$t_{f}$ = 30.10 mm < 100~mm
  \item[]Use buckling curve B,  $\alpha_{y}$ = 0.34 --- (EN 1993-1-1 :2022, Table 8.2)
  \item[]$\phi_{y}$ = $0.5 \cdot (1 + \alpha_{y} \cdot (\lambda_{y} - 0.2) + {\lambda_{y}}^{2})$ = 0.93 --- (EN 1993-1-1 :2022, Eq 8.74)
  \item[]$\chi_{y}$ = $\frac{1}{\phi_{y} + \sqrt{{\phi_{y}}^{2} - {\lambda_{y}}^{2}}}$ = 0.72 --- (EN 1993-1-1 :2022, Eq 8.73)
\end{description}

\section{}\label{section-11}

\section{Lateral torsional buckling
check}\label{lateral-torsional-buckling-check}

\begin{description}
    \item[]Critical moment, $M_{cr}$= $\frac{\pi^{2} \cdot E \cdot I_{z}}{L_{cr}^{2}} \cdot \left( \frac{I_{w}}{I_{z}} + \frac{G \cdot I_{t} \cdot L_{cr}^{2}}{\pi^{2} \cdot E \cdot I_{t}} \right)$ = 3373.12 kNm
    \item[]For $\frac{h}{b}$ > 1.2 and $t_{f}$ <= 40 mm, 
    \item[]$\alpha_{LT}$ = $\min(0.12\cdot \sqrt{\frac{W_{el,y}}{W_{el,z}}}, 0.34)$ = 0.27 --- (EN 1993-1-1 :2022, Table 8.5)
    \item[]$\phi_{LT}$ = $0.5\cdot(1 + \alpha_{LT}\cdot(\lambda_z - 0.2) + \lambda_z^{2})$ = 0.91 --- (EN 1993-1-1 :2022, Eq 8.82)
    \item[]$\chi_{LT}$ = $\min(\frac{1}{\phi_{LT} + \sqrt{\phi_{LT}^{2} - \lambda_z^{2}}},1)$ = 0.76 --- (EN 1993-1-1 :2022, Eq 8.81)
    \item[]Moment resistance (y-axis), $M_{brd}$ = $\frac{\chi_{LT} \cdot W_{pl,y} \cdot f_{y}}{\gamma_1}$ = 409.37 kNm --- (EN 1993-1-1 :2022, Eq 8.79)
    \item[]Design moment about y-axis, $M_{yed}$ = 53.57 kNm
    \item[]Moment resistance (z-axis), $M_{crd,z}$ = $\frac{W_{pl,z} \cdot f_{y}}{\gamma_1}$ = 249.92 kNm --- (EN 1993-1-1 :2022, Eq 8.20)
    \item[]Design moment about z-axis, $M_{zed}$ = 30.00 kNm
    \item[]Therefore, the section is safe in resisting bending
\end{description}

\section{Axial and bending interaction
check}\label{axial-and-bending-interaction-check}

\begin{description}
    \item[]Assume simple construction and refer to EN 1993-1-1 :2022, Eq 8.59 
    \item[]$\frac{N_{ed}}{N_{rk}} + \frac{M_{y}}{M_{brd}} + \frac{M_{z}}{M_{cr,dz}}$ = 2.37
    \item[]Therefore, the result is greater than 1\, the section is not safe in resisting interaction
\end{description}
